import { Stack, Text } from "~components/dashboard";

## Stack

### Basic Usage

<Stack {...props}>
  <h1>Example</h1>
  <Text>Stack demonstration</Text>
</Stack>

#### Configuration

```json
{
  "children": [
    {
      "type": "stack",
      "properties": {
        "direction": "vertical"
      },
      "children": [
        {
          "type": "text",
          "properties": {
            "value": "Example"
          },
        },
        {
          "type": "text",
          "properties": {
            "value": "Stack demonstration"
          }
        }
      ]
    }
  ]
}
```
