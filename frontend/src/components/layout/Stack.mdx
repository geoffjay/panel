import { Stack } from "~components/layout";

# Stack

## Basic Usage

<Stack {...props}>
  <h1>Example</h1>
  <p>Stack demonstration</p>
</Stack>

### Configuration

```yaml
layout:
  components:
    - type: stack
      direction: vertical
      children:
        - type: text
          value: Example
        - type: text
          value: Stack demonstration
```
